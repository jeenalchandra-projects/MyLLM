class AttentionHead(nn.Module):
  def forward(self, x):
    q = self.query(x)
    k = self.key(x)
    v = self.value(x)
    # scaled dot-product attention
    scores = q @ k.transpose(-2,-1)
    scores /= head_size ** 0.5
    weights = softmax(scores)
    return weights @ v
